// SESSION: Start / Switch / Stop

async function startStream() {
  // ขอ stream → user เลือก Window/Tab/Screen
  const stream = await navigator.mediaDevices.getDisplayMedia({
    video: { cursor: 'always', frameRate: 30 },
    audio: false,
  });
  // listen เมื่อ user หยุดแชร์เอง
  stream.getVideoTracks()[0].addEventListener('ended', stopSession);
  return stream;
}

els.startBtn.addEventListener('click', async () => {
  try {
    const stream = await startStream();
    state.stream = stream;
    els.preview.srcObject = stream;
    await els.preview.play();

    // รอ video metadata โหลด
    await new Promise(r => {
      if (els.preview.videoWidth) r();
      else els.preview.addEventListener('loadedmetadata', r, { once: true });
    });

    initCropFrame();

    els.previewPanel.classList.remove('hidden');

// CROP FRAME: Init / Drag / Resize

function getPreviewBox() {
  // Effective video display area inside <video>
  const wrap = els.preview.getBoundingClientRect();
  const vW = els.preview.videoWidth, vH = els.preview.videoHeight;
  if (!vW || !vH) return { left: 0, top: 0, width: wrap.width, height: wrap.height };
  // object-fit: contain → letterbox
  const scale = Math.min(wrap.width / vW, wrap.height / vH);
  const dispW = vW * scale, dispH = vH * scale;
  const left = (wrap.width - dispW) / 2;
  const top = (wrap.height - dispH) / 2;
  return { left, top, width: dispW, height: dispH };
}

function initCropFrame() {
  const box = getPreviewBox();
  if (state.ratio.free) {
    state.crop = { x: box.left, y: box.top, w: box.width, h: box.height };
  } else {
    const ratio = state.ratio.w / state.ratio.h;
    let w = box.width * 0.7;
    let h = w / ratio;
    if (h > box.height * 0.9) { h = box.height * 0.9; w = h * ratio; }
    state.crop = {
      x: box.left + (box.width - w) / 2,
      y: box.top + (box.height - h) / 2,
      w, h,
    };
  }
  els.ratioLabel.textContent = state.ratio.free ? 'Full' : `${state.ratio.w}:${state.ratio.h}`;
  updateFrame();
}

function updateFrame() {
  els.cropFrame.style.left = state.crop.x + 'px';
  els.cropFrame.style.top = state.crop.y + 'px';
  els.cropFrame.style.width = state.crop.w + 'px';
  els.cropFrame.style.height = state.crop.h + 'px';
}

function clampToPreview() {
  const box = getPreviewBox();
  if (state.crop.w > box.width) state.crop.w = box.width;
  if (state.crop.h > box.height) state.crop.h = box.height;
  if (state.crop.x < box.left) state.crop.x = box.left;
  if (state.crop.y < box.top) state.crop.y = box.top;
  if (state.crop.x + state.crop.w > box.left + box.width) state.crop.x = box.left + box.width - state.crop.w;
  if (state.crop.y + state.crop.h > box.top + box.height) state.crop.y = box.top + box.height - state.crop.h;
}

let dragMode = null;
let dragStart = null;

els.cropFrame.addEventListener('mousedown', (e) => {

  e.preventDefault();
  dragMode = 'move';

document.querySelectorAll('.handle').forEach(h => {
  h.addEventListener('mousedown', (e) => {
    e.preventDefault();
    e.stopPropagation();
    dragMode = 'resize-' + h.dataset.pos;
    dragStart = {
      mx: e.clientX, my: e.clientY,
      cx: state.crop.x, cy: state.crop.y,
      cw: state.crop.w, ch: state.crop.h,
    };
  });

document.addEventListener('mousemove', (e) => {
  if (!dragMode || !dragStart) return;
  const dx = e.clientX - dragStart.mx;
  const dy = e.clientY - dragStart.my;

  if (dragMode === 'move') {
    state.crop.x = dragStart.cx + dx;
    state.crop.y = dragStart.cy + dy;
  } else {
    const ratio = state.ratio.free ? (dragStart.cw / dragStart.ch) : (state.ratio.w / state.ratio.h);
    const pos = dragMode.split('-')[1];
    let newW = dragStart.cw, newH = dragStart.ch;
    let newX = dragStart.cx, newY = dragStart.cy;

    if (pos === 'se') {
      newW = dragStart.cw + dx;
      if (!state.ratio.free) newH = newW / ratio;
      else newH = dragStart.ch + dy;
    } else if (pos === 'sw') {
      newW = dragStart.cw - dx;
      if (!state.ratio.free) newH = newW / ratio;
      else newH = dragStart.ch + dy;
      newX = dragStart.cx + (dragStart.cw - newW);
    } else if (pos === 'ne') {
      newW = dragStart.cw + dx;
      if (!state.ratio.free) newH = newW / ratio;
      else newH = dragStart.ch - dy;
      newY = dragStart.cy + (dragStart.ch - newH);
    } else if (pos === 'nw') {
      newW = dragStart.cw - dx;
      if (!state.ratio.free) newH = newW / ratio;
      else newH = dragStart.ch - dy;
      newX = dragStart.cx + (dragStart.cw - newW);
      newY = dragStart.cy + (dragStart.ch - newH);
    }
    if (newW < 50 || newH < 50) return;
    state.crop.w = newW;
    state.crop.h = newH;
    state.crop.x = newX;
    state.crop.y = newY;
  }
  clampToPreview();
  updateFrame();
});

document.addEventListener('mouseup', () => {

  dragStart = null;
});

// ════════════════════════════════════════════════════════════
// CAPTURE (heart of the system)
// ════════════════════════════════════════════════════════════
function capture() {
  if (!state.stream || !els.preview.videoWidth) {
    toast('⚠️ Start session ก่อน');
    return;
  }
  const vW = els.preview.videoWidth;
  const vH = els.preview.videoHeight;
  const box = getPreviewBox();

  // CSS coords (within preview wrap) → video pixel coords
  const scaleX = vW / box.width;
  const scaleY = vH / box.height;
  const sx = (state.crop.x - box.left) * scaleX;
  const sy = (state.crop.y - box.top) * scaleY;
  const sw = state.crop.w * scaleX;
  const sh = state.crop.h * scaleY;

  const canvas = document.createElement('canvas');
  canvas.width = Math.round(sw);
  canvas.height = Math.round(sh);
  const ctx = canvas.getContext('2d');
  ctx.drawImage(els.preview, sx, sy, sw, sh, 0, 0, canvas.width, canvas.height);

  const slide = {
    id: Date.now() + Math.random(),
    dataUrl: canvas.toDataURL('image/png'),
    timestamp: new Date().toLocaleTimeString(),
    width: canvas.width,
    height: canvas.height,
  };
  state.slides.push(slide);
  renderSlides();
  updatePipCount();
  toast(`📸 Captured #${state.slides.length}`);

  // Flash effect
  els.cropFrame.style.background = 'rgba(255,255,255,0.4)';
  setTimeout(() => { els.cropFrame.style.background = ''; }, 100);

  // Beep
  try {
    const ac = new (window.AudioContext || window.webkitAudioContext)();
    const o = ac.createOscillator(); const g = ac.createGain();
    o.connect(g); g.connect(ac.destination);
    o.frequency.value = 880; g.gain.setValueAtTime(0.1, ac.currentTime);
    g.gain.exponentialRampToValueAtTime(0.001, ac.currentTime + 0.1);
    o.start(); o.stop(ac.currentTime + 0.1);
  } catch(e){}
}

els.captureBtn.addEventListener('click', capture);

  if (e.code === 'Space' && state.stream && !e.target.matches('input,textarea')) {

  }
});

      body { margin:0; font-family:Segoe UI,sans-serif; background:#1f2937; color:#fff; padding:12px; }
      h3 { font-size:0.9em; margin-bottom:8px; opacity:0.8; }
      button { width:100%; padding:18px; font-size:1.1em; font-weight:700;
               background:#10b981; color:#fff; border:none; border-radius:10px;
               cursor:pointer; margin-bottom:8px; transition:0.2s; }
      button:hover { background:#059669; transform:scale(1.02); }
      button:active { transform:scale(0.98); }
      .count { text-align:center; font-size:1.5em; font-weight:700; color:#fbbf24; }
      .label { text-align:center; font-size:0.75em; opacity:0.6; }
      .hint { font-size:0.7em; opacity:0.5; text-align:center; margin-top:6px; }
    </style>`;
    pip.document.body.innerHTML = `
      <h3>📸 Snipper Floating</h3>
      <button id="pipCap">📸 CAPTURE (F5)</button>
      <div class="label">Total slides</div>
      <div class="count" id="pipCount">0</div>
      <div class="hint">F5 / Space / Click</div>
    `;
    pip.document.getElementById('pipCap').addEventListener('click', capture);
    pip.document.addEventListener('keydown', (e) => {
      if (e.key === 'F5' || e.code === 'Space') {
        e.preventDefault();
        capture();
      }
    });
    pip.addEventListener('pagehide', () => { state.pipWindow = null; });
    updatePipCount();
    toast('🪟 Floating window opened — pin ไว้เลย!', 2500);
  } catch (err) {
    toast('❌ ' + err.message, 3000);
  }
});

function updatePipCount() {
  if (state.pipWindow) {
    const el = state.pipWindow.document.getElementById('pipCount');
    if (el) el.textContent = state.slides.length;
  }
}

// ════════════════════════════════════════════════════════════
// SLIDES RENDER / CLEAR / EXPORT
// ════════════════════════════════════════════════════════════
function renderSlides() {
  els.count.textContent = state.slides.length;
  els.exportBtn.disabled = state.slides.length === 0;
  if (state.slides.length === 0) {
    els.slidesContainer.innerHTML = `<div class="empty-state"><div class="icon">📷</div><p>ยังไม่มี slide</p></div>`;
    return;
  }
  const grid = document.createElement('div');
  grid.className = 'slides-grid';
  state.slides.forEach((s, i) => {
    const card = document.createElement('div');
    card.className = 'slide-card';
    card.innerHTML = `
      <img src="${s.dataUrl}" alt="Slide ${i+1}">
      <div class="slide-info">
        <span>#${i+1} • ${s.timestamp}</span>
        <button class="slide-del">×</button>
      </div>`;
    card.querySelector('.slide-del').addEventListener('click', () => {
      state.slides = state.slides.filter(x => x.id !== s.id);
      renderSlides();
      updatePipCount();
    });
    grid.appendChild(card);
  });
  els.slidesContainer.innerHTML = '';
  els.slidesContainer.appendChild(grid);
}

els.clearBtn.addEventListener('click', () => {
  if (state.slides.length === 0) return;
  if (confirm(`ลบทั้งหมด ${state.slides.length} slides?`)) {
    state.slides = [];
    renderSlides();
    updatePipCount();
    toast('🗑️ Cleared');
  }
});

els.exportBtn.addEventListener('click', async () => {
  if (state.slides.length === 0) return;
  toast('⏳ Exporting...');
  const pptx = new PptxGenJS();
  const first = state.slides[0];
  const r = first.width / first.height;
  if (Math.abs(r - 16/9) < 0.05) pptx.layout = 'LAYOUT_16x9';
  else if (Math.abs(r - 4/3) < 0.05) pptx.layout = 'LAYOUT_4x3';
  else {
    pptx.defineLayout({ name: 'CUSTOM', width: 10, height: 10/r });
    pptx.layout = 'CUSTOM';
  }
  state.slides.forEach(s => {
    const slide = pptx.addSlide();
    slide.background = { color: '000000' };
    slide.addImage({
      data: s.dataUrl, x: 0, y: 0, w: '100%', h: '100%',
      sizing: { type: 'contain', w: '100%', h: '100%' },
    });
  });
  await pptx.writeFile({ fileName: `Snipper_${new Date().toISOString().slice(0,10)}_${Date.now()}.pptx` });
  toast(`✅ Exported ${state.slides.length} slides!`, 3000);
});
